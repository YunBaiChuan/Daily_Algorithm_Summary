## 第一题：3583.统计特殊三元组 https://leetcode.cn/problems/count-special-triplets/?envType=daily-question&envId=2025-12-09

**思路：比较经典的思路，枚举中间，然后左右两边的2 * x元素个数乘积累加和并且不断取mod即可。优化也是比较经典的两个前缀和优化，不过注意一下有个恢复现场以及内存优化(每次不全部算出suf，只算需要的2 * x数量即可)**

**代码：**
```python
class Solution:
    def specialTriplets(self, nums: List[int]) -> int:
        mod = 10 ** 9 + 7 
        ans = 0
        n = len(nums)
        total = Counter(nums)
        pre = Counter([nums[0]])
        for j in range(1, n - 1):
            x = nums[j]
            total[x] -= 1
            a, b = pre[2 * x], total[2 * x] - pre[2 * x]
            ans = (ans + a * b) % mod 
            pre[x] += 1
            total[x] += 1
        return ans
```

import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
有向树，dp
'''

def solve():
    n = II()
    w = LII()
    g = [[] for _ in range(n + 1)]
    is_sub = [False] * (n + 1)  # 标记是否做过“子节点”

    for _ in range(n - 1):
        k, l = MII()
        g[k].append(l)
        is_sub[l] = True

    # 找根：唯一没当过子节点的就是总 boss
    root = next(i for i in range(1, n + 1) if not is_sub[i])

    # dp[u][0] 不选 u 的最大值
    # dp[u][1] 选 u 的最大值
    dp = [[0, 0] for _ in range(n+1)]

    def dfs(u):
        dp[u][1] = w[u-1]
        dp[u][0] = 0
        for v in g[u]:
            dfs(v)
            dp[u][0] += max(dp[v][0], dp[v][1])
            dp[u][1] += dp[v][0]    # 选 u 就不能选直接下属

    dfs(root)
    print(max(dp[root][0], dp[root][1]))

t = 1
# t = II()
for _ in range(t):
    solve()
