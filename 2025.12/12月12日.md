```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18


def I():
    return input()


def II():
    return int(input())


def MII():
    return map(int, input().split())


def GMI():
    return map(lambda x: int(x) - 1, input().split())


def LI():
    return input().split()


def LII():
    return list(map(int, input().split()))


def LFI():
    return list(map(float, input().split()))


fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
每个位置可以这样改任意大写、小写字母、数字、特殊符号
26、26、10、4
'''

def solve():
    s = I()

    cnt = [0] * 4
    for c in s:
        if 'A' <= c <= 'Z':
            cnt[0] += 1
        elif 'a' <= c <= 'z':
            cnt[1] += 1
        elif c.isdigit():
            cnt[2] += 1
        else:
            cnt[3] += 1

    ans = 0
    for i, x in enumerate(cnt):
        if x != 1:
            ans += x * 65
        else:
            if i == 0 or i == 1:
                ans += 25
            elif i == 2:
                ans += 9
            else:
                ans += 3

    print(ans)

# t = 1
t = II()
for _ in range(t):
    solve()
```
