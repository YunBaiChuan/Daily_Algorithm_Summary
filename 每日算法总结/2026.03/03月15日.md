**思路：多重背包板子题，一般就是两种方法：1.二进制分组 + 01背包（好写，但复杂度稍高）；2.余数分组 + 单调队列优化（难写点，复杂度最优）**

**代码：二进制分组 + 01背包**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

out = []

def solve():
    m, n = MII()
    groups = []
    for _ in range(m):
        v, w, cnt = LII()
        res = cnt
        for i in range(cnt):
            j = 1 << i
            if res >= j:
                groups.append((j * v, j * w))
                res -= j
        if res > 0:
            groups.append((res * v, res * w))

    l = len(groups)
    dp = [0] * (n + 1)
    for i in range(l):
        v, w = groups[i]
        for j in range(n, -1, -1):
            if j >= v:
                dp[j] = fmax(dp[j], dp[j - v] + w)

    out.append(str(dp[-1]))

# t = 1
t = II()
for _ in range(t):
    solve()

print('\n'.join(out))
```
