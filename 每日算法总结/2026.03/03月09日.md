## 第一题：小苯的蓄水池 https://www.nowcoder.com/practice/06499d54e87d46569385652b2f954752?channelPut=tracker2

**思路：比较容易看出来是并查集，不过合并区间那里容易超时，然后递归深度也可能超。我们就需要额外引入一个跳表来记录右侧第一个还没合并的位置，就可以实现快速合并；再把递归改成栈模拟，就能解决递归超深度问题**

**代码：**

import sys

import math

    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

class UnionFind:
    def __init__(self, n, a):
        self.fa = list(range(n))
        self.size = [1] * n
        self.a = [0] + a
        self.sets = n

    def find(self, x):
        root = x
        while root != self.fa[root]:
            root = self.fa[root]
        while x != root:
            self.fa[x], x = root, self.fa[x]
        return root

    def is_same(self, x, y):
        fx, fy = self.find(x), self.find(y)
        return fx == fy

    def union(self, x, y):
        fx, fy = self.find(x), self.find(y)
        if fx == fy:
            return False
        self.fa[fx] = fy
        self.size[fy] += self.size[fx]
        self.a[fy] += self.a[fx]
        self.sets -= 1
        return True

def solve():
    n, m = MII()
    a = LII()

    uf = UnionFind(n + 1, a)
    nxt = list(range(n + 2))

    def find_nxt(x):
        if x > n:
            return x
        root = x
        while nxt[root] != root:
            root = nxt[root]
        while x != root:
            nxt[x], x = root, nxt[x]
        return root

    out = []
    for _ in range(m):
        op, *nums = LII()
        if op == 1:
            l, r = nums
            i = find_nxt(l)
            while i < r:
                uf.union(i, i + 1)
                nxt[i] = i + 1
                i = find_nxt(i + 1)
        else:
            x = nums[0]
            fx = uf.find(x)
            a, b = uf.a[fx], uf.size[fx]
            out.append(f"{a / b:.10f}")

    print('\n'.join(out))

t = 1
for _ in range(t):
    solve()
