## 第一题：空调遥控 https://www.nowcoder.com/practice/7cb58d56b30c4280ac07ae4428022e03?channelPut=tracker2

**思路：排序 + 滑动窗口。主要是把绝对值这个条件进行转化，变成右 - 左 > 2 * p时进行窗口的移动，那么结果其实就是越短越合法，最终不断取max就是答案了**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, p = MII()
    a = LII()
    a.sort()

    ans = l = 0
    for r in range(n):
        while a[r] - a[l] > 2 * p:
            l += 1
        ans = fmax(ans, r - l + 1)

    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```
