**思路：比较新颖的题目，主要是引入了额外的枚举，用于固定状态数，接着再用一般的dp + 枚举，不断更新答案**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    s = I()
    n = len(s)
    c0 = 0
    cq = 0
    for c in s:
        if c == 'o':
            c0 += 1
        elif c == '?':
            cq += 1

    ans = 0
    for i in range(c0, c0 + cq + 1): # 枚举o数量
        dp = [[-inf] * (i + 1) for _ in range(n + 1)] # dp[i][j]表示前i个位置，有j个o时的最大ovo子序列数量
        dp[0][0] = 0
        for j in range(1, n + 1):
            c = s[j - 1]
            for k in range(i + 1):
                if c != 'o' and dp[j - 1][k] != -inf:
                    dp[j][k] = fmax(dp[j][k], dp[j - 1][k] + k * (i - k))
                if c != 'v' and k > 0 and dp[j - 1][k - 1] != -inf:
                    dp[j][k] = fmax(dp[j][k], dp[j - 1][k - 1])
        ans = fmax(ans, dp[n][i])

    print(ans)


# t = 1
t = II()
for _ in range(t):
    solve()
```
