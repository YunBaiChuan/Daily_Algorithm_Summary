**思路：本题需要将问题转化为图论，接着就是用度数为1的节点进行dfs匹配。匹配完成后，如果有节点度数为0，说明不合法，否则收集匹配完成后的答案。最终，输出即可**

**代码：**

import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

def solve():
    n = II()
    graph = [[] for _ in range(n + 1)]
    du = [0] * (n + 1)

    vise = [False] * n
    vis = [False] * (n + 1)
    res = [0] * n

    for i in range(n):
        tmp = LII()
        k = tmp[0]
        p1 = tmp[1]
        if k == 1:
            p2 = p1
        else:
            p2 = tmp[2]

        du[p1] += 1
        du[p2] += 1
        graph[p1].append((p2, i))
        graph[p2].append((p1, i))

    def dfs(x):
        vis[x] = True
        for y, v in graph[x]:
            if not vise[v]:
                vise[v] = True
                du[x] -= 1
                du[y] -= 1
                res[v] = x
                if du[y] == 1:
                    dfs(y)
                return

    for i in range(1, n + 1):
        if du[i] == 1 and not vis[i]:
            dfs(i)

    for i in range(1, n + 1):
        if not vis[i]:
            if du[i] == 0:
                print("kou is angry")
                return
            dfs(i)

    print(' '.join(map(str, res)))

t = 1
# t = II()
for _ in range(t):
    solve()
