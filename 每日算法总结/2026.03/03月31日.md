## 第一题：最大最小路 https://www.nowcoder.com/practice/bce59fc8643b47359e9521c7e590b239?channelPut=tracker2

**思路：并查集 + 容斥原理。每个连通块中，假设其大小为k，那么可以连通的路径数量为$k * (k + 1)\ //\ 2$。因此，我们先算出整体的路径条数，即$n * (n + 1)\ //\ 2$，接着再减去$ > a$和$< b$路径数，最终再加上$a <= x <= b$的路径数，就可以得到最终答案了**

**实现方面可以写一个cal计算函数，这样就能方便三次的计算；还有就是如何排除不要的端点值？我们可以直接从一开始就排除掉，不让他们参与运算**

class UnionFind:
    def __init__(self, n):
        self.fa = list(range(n))
        self.size = [1] * n

    def find(self, x):
        root = x
        while root != self.fa[root]:
            root = self.fa[root]
        while x != root:
            parent = self.fa[x]
            self.fa[x] = root
            x = parent
        return root

    def is_same(self, x, y):
        return self.find(x) == self.find(y)

    def union(self, x, y):
        fx, fy = self.find(x), self.find(y)
        if fx == fy:
            return False
        self.fa[fx] = fy
        self.size[fy] += self.size[fx]
        return True

def cal(n, edges, w, l, r):
    uf = UnionFind(n + 1)
    valid = [0] + [l <= w[i] <= r for i in range(1, n + 1)]

    for u, v in edges:
        if valid[u] and valid[v]:
            uf.union(u, v)

    roots = []
    for i in range(1, n + 1):
        if valid[i]:
            roots.append(uf.find(i))

    cnt = Counter(roots)
    res = 0
    for num in cnt.values():
        res += num * (num + 1) // 2
    return res

def solve():
    n, a, b = LII()
    w = [0] + LII()

    edges = []
    for _ in range(n - 1):
        x, y = MII()
        edges.append((x, y))

    total = n * (n + 1) // 2
    ra = cal(n, edges, w, a + 1, inf)
    rb = cal(n, edges, w, -inf, b - 1)
    rab = cal(n, edges, w, a + 1, b - 1)

    ans = total - ra - rb + rab # 容斥原理
    print(ans)

t = 1
