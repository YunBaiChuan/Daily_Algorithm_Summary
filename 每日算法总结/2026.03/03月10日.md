## 第一题：不是烤串故事 https://www.nowcoder.com/practice/d63eb724bcde43639dc676abf2f9fd81?channelPut=tracker2

**思路：字符串z函数。z函数的$z[i]$表示$s[i:]$与$s[0:]$的最长公共前缀长度**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

out = []

def zf(s):
    n = len(s)
    z = [0] * n
    l = r = 0
    for i in range(1, n):
        if i <= r:
            z[i] = fmin(r - i + 1, z[i - l])
        while i + z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l = i
            r = i + z[i] - 1
    return z

def solve():
    n = II()
    s = I().encode()
    t = I().encode()

    rs = s[::-1]
    sep = b'{'

    # Z函数: t 与 rs 的匹配
    # z[i] 表示 s[i:] 与 s[0:] 的最长公共前缀长度
    z = zf(t + sep + rs)
    lrs = z[n + 1:]

    lcp = [0] * (n + 1)
    for i in range(n - 1, -1, -1):
        if s[i] == t[i]:
            lcp[i] = lcp[i + 1] + 1

    best = -1
    besti = 1

    for i0 in range(1, n + 1):
        first = lrs[n - i0]
        if first > i0:
            first = i0 

        if first < i0:
            l = first
        else:
            l = i0 + lcp[i0]

        if l > best or (l == best and i0 < besti):
            best = l
            besti = i0

    out.append(f"{best} {besti}")

# tt = 1
tt = II()
for _ in range(tt):
    solve()

print('\n'.join(out))
```
