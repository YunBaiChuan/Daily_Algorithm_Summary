## 第一题：执行操作后的最小汉明距离 https://leetcode.cn/problems/minimize-hamming-distance-after-swap-operations/description/?envType=daily-question&envId=2026-04-21

**思路：并查集 + hash表。对于这种交换问题来说，其实就是能够交换的元素，就是一个连通块，那么连通块就可以用dfs或者并查集来做，我比较喜欢用并查集。最终统计的时候需要注意一下，直接统计汉明距离其实是不好算的，因此我们逆向思考，先统计匹配的个数，这个其实就是hash表的cnt次数取min，最终用总长度减去匹配个数，就是当前分组的答案，把他累加到最终ans中即可**

**代码：**
```python
class UnionFind:
    def __init__(self, n):
        self.fa = list(range(n))
        self.size = [1] * n
        self.sets = n

    def find(self, x):
        if x != self.fa[x]:
            self.fa[x] = self.find(self.fa[x])
        return self.fa[x]

    def is_same(self, x, y):
        fx, fy = self.find(x), self.find(y)
        return fx == fy

    def union(self, x, y):
        fx, fy = self.find(x), self.find(y)
        if fx == fy:
            return False
        self.fa[fx] = fy
        self.size[fy] += self.size[fx]
        self.sets -= 1
        return True

class Solution:
    def minimumHammingDistance(self, source: List[int], target: List[int], allowedSwaps: List[List[int]]) -> int:
        n = len(source)
        if not allowedSwaps:
            cnt = 0 
            for i in range(n):
                cnt += (source[i] != target[i])
            return cnt 
        
        uf = UnionFind(n)
        for u, v in allowedSwaps:
            if not uf.is_same(u, v):
                uf.union(u, v)
        
        groups = [[] for _ in range(n)] # 连通块分组
        for i, x in enumerate(uf.fa):
            fx = uf.find(x)
            groups[fx].append(i)
        
        ans = 0 
        for i, row in enumerate(groups):
            if not row:
                continue 
                
            res = 0
            cnt1, cnt2 = Counter(), Counter()
            for j, x in enumerate(row):
                cnt1[source[x]] += 1 
                cnt2[target[x]] += 1
            
            for k, v in cnt1.items(): # 匹配个数
                res += min(v, cnt2[k])
            
            ans += len(row) - res # 不匹配个数（汉明距离）
        return ans 
```
