import sys
from functools import lru_cache

input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''


def solve():
    n, m = MII()
    g = [[] for _ in range(n + 1)]
    for _ in range(m):
        u, v, w = LII()
        g[u].append((v, w))

    dp = [0.0] * (n + 1)
    @lru_cache(maxsize=None)
    def dfs(u):
        if u == n:
            return 0.0
        k = len(g[u])
        total = 0.0
        for v, w in g[u]:
            total += w + dfs(v)
        dp[u] = total / k
        return dp[u]

    ans = dfs(1)
    print(f"{ans:.2f}")

t = 1
# t = II()
for _ in range(t):
    solve()
