**思路：题目要求最大化MAX - MEX，也就是说MAX尽量大，MEX尽量小，并且限制了区间长度至少为2。因此，我们直接去检查长度为2的区间，他们的MAX - MEX，然后不断取最大，就是最终的答案了**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n = II()
    a = LII()

    ans = -inf
    for i in range(n - 1):
        mex = 0
        maxn = fmax(a[i], a[i + 1])
        while mex == a[i] or mex == a[i + 1]:
            mex += 1
        ans = fmax(ans, maxn - mex)

    print(ans)

# t = 1
t = II()
for _ in range(t):
    solve()
```
