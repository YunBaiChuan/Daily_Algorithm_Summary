**思路：大方向是恰好滑动窗口类型，但是结尾0不具有单调性，因此需要进行转化，那该如何转化？考虑后缀0，那就应该能被10整除，因此就可以进行因子的分解，也就是$10^k = (2 * 5)^k = 2^k * 5^k$，那就提前预处理出每个数对应的因子个数，这样窗口就具备了单调性，然后就可以跑一个恰好滑动窗口了**

**代码：**

import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

def solve():
    n, k = MII()
    a = LII()

    cnt2 = []
    cnt5 = []
    for x in a:
        c2 = c5 = 0
        while x % 2 == 0:
            c2 += 1
            x //= 2
        while x % 5 == 0:
            c5 += 1
            x //= 5
        cnt2.append(c2)
        cnt5.append(c5)

    def func(k):
        if k == 0:
            return n * (n + 1) // 2

        ans = 0
        sum2 = sum5 = 0
        l = 0

        for r in range(n):
            sum2 += cnt2[r]
            sum5 += cnt5[r]
            while sum2 >= k and sum5 >= k:
                sum2 -= cnt2[l]
                sum5 -= cnt5[l]
                l += 1
            ans += l

        return ans

    ans = func(k) - func(k + 1)
    print(ans)


t = II()
for _ in range(t):
    solve()
