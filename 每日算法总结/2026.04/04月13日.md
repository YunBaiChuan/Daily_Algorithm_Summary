**思路：思维题。有点类似01串翻转问题，不过01串翻转问题是相邻进行操作，本题是中间隔了一个元素，然后有个+1 -1或者-1 +1。这类问题的通用思考是多次向后滑动操作，规律就显现出来了。本题就是分成两组，这两组元素相同，那就分别根据相等元素的差值进行遍历，最终判断差值是否为0即可**

**代码：**

import sys
input = lambda: sys.stdin.readline().strip()

import math

def I():
    return input()

def II():
    return int(input())

    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

def solve():
    n = II()
    a = LII()

    if sum(a) % n != 0:
        print("NO")
        return

    rq = sum(a) // n
    ans1 = 0
    for i in range(0, n, 2):
        ans1 += a[i] - rq
    ans2 = 0
    for i in range(1, n, 2):
        ans2 += a[i] - rq

    print("YES" if ans1 == ans2 == 0 else "NO")

# t = 1
t = II()
for _ in range(t):
    solve()
