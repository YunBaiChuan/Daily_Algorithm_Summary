import sys
from collections import defaultdict

input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y

def solve():
    n = II()
    d = defaultdict(int)
    ans = 0
    st = set()
    for _ in range(n):
        s = I()
        d[s] += 1
        if s not in st and d[s] == ans + 1:
            ans += 1
            st.add(s)
    print(ans)

t = 1

    solve()
