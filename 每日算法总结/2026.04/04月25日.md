**思路：这题不能用01背包板子，因为&没有+那样的单调性，状态转移写不出来。因此，本题需要用位枚举 + 贪心，从高位到低位进行枚举，然后判断能够达到答案**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, k = MII()
    a = [LII() for _ in range(n)]

    ans = 0
    for mask in range(10, -1, -1):
        target = ans | (1 << mask)
        v_and = (1 << 11) - 1
        ok = False

        for i in range(n):
            v, w = a[i]
            if w & target == target: # 能选的都选，可以把体积变小，并且当前位的结果依然最优
                v_and &= v
                ok = True

        if ok and v_and <= k:
            ans = target

    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```
