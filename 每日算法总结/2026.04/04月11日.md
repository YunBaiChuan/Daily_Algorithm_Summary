## 第一题：圆覆盖 https://www.nowcoder.com/practice/4f96afe5dfe74dad88dbe419d33f9536?channelPut=tracker2

**思路：半径具有单调性，半径越大，覆盖的面积越大，越容易满足条件；否则半径越小，越不容易满足条件。那就可以用二分，对于本题有要求的精度来说，就需要用到浮点数二分**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, s = MII()
    a = []
    total = 0
    mx = 0.0
    for _ in range(n):
        x, y, v = LII()
        a.append((x, y, v))
        mx = fmax(mx, math.sqrt(x * x + y * y))
        total += v

    if total < s:
        print(-1)
        return

    def check(mid):
        ans = 0
        for x, y, v in a:
            if x * x + y * y <= mid * mid:
                ans += v
        return ans >= s

    l, r = 0.0, mx
    while r - l > 1e-6: # for _ in range(100):
        mid = (l + r) / 2.0
        if check(mid):
            r = mid
        else:
            l = mid
    print(l)

t = 1
# t = II()
for _ in range(t):
    solve()
```
