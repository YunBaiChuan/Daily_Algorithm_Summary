**思路：数学，推式子。首先要知道一个公式：$gcd(a + c,\ b + c) = gcd(a + c,\ b - a)$，因此我们可以令$abs(b - a) = d$，然后就是根据$gcd(a + c, d) > 1$来算c了。具体怎么计算呢？有一个数学结论，他是对称的。枚举d的质因子p，a + c中必然包含该质因子，此时$gcd(a + c, d) > 1$，然后就可以根据余数式子来推c的表达式为：$a + c = 0 (mod\ p)\ -> c = -a (mod\ p) -> c = (p - a\ mod\ p)\ mod\ p$，然后不断取fmin的c，就能得到最终答案了**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
gcd(a + c, b + c) = gcd(a + c, b - a) != 1
'''

def gcd(a, b):
    return a if b == 0 else gcd(b, a % b)

def solve():
    a, b = MII()

    if gcd(a, b) > 1:
        print(0)
        return

    d = abs(a - b)

    if d == 0:
        print(1)
        return

    if d == 1:
        print(-1)
        return

    ans = inf
    for i in range(2, isqrt(d) + 1):
        if d % i == 0:
            g1 = i
            c = (g1 - a % g1) % g1
            ans = fmin(ans, c)

            g2 = d // i
            if g2 > 1:
                c = (g2 - a % g2) % g2
                ans = fmin(ans, c)

            while d % i == 0:
                d //= i

    if d > 1:
        c = (d - a % d) % d
        ans = fmin(ans, c)

    print(ans if ans != inf else -1)

t = 1
# t = II()
for _ in range(t):
    solve()
```
