**思路：非常裸的状态dp板子，但是要注意下初始化应该都是-inf，表示初始状态不可达，而并不是0。还有就是，如果当前钱不够k的时候，不能进行穿越转移。这题挺典型的，可以放到板子题里面**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, k = MII()
    a = LII()
    b = LII()
    dp = [[-inf] * 2 for _ in range(n + 1)]
    dp[1][0] = a[0]
    for i in range(1, n):
        dp[i + 1][0] = fmax(dp[i][0], dp[i][1] - k) + a[i] \
                            if dp[i][1] >= k else dp[i][0] + a[i]
        dp[i + 1][1] = fmax(dp[i][1], dp[i][0] - k) + b[i] \
                            if dp[i][0] >= k else dp[i][1] + b[i]
    print(fmax(dp[n][0], dp[n][1]))

t = 1
# t = II()
for _ in range(t):
    solve()
```
