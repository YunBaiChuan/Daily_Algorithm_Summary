**思路：网格图dfs + 矩形判断。这题网格图dfs好写，不过要注意group参数放的位置，可以不放在递归参数上，写起来比较方便，如果放在参数上需要注意位置和返回值；然后就是矩形判断，这个得好好学习一下，其实也不难：1.分别收集行索引和列索引，并分别得到对应的最大值和最小值；2.判断点的个数是否和前面通过计算最大值和最小值得到的面积相同。为什么可行？因为矩形的点是连续的，因此用点的数目和面积的大小进行判断，是可行且简洁的**

import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

def solve():
    n, m = MII()
    a = [list(I()) for _ in range(n)]

    groups = []
    for i in range(n):
        for j in range(m):
            if a[i][j] != '.':
                continue

            stack = [(i, j)]
            a[i][j] = '*'
            group = []

            while stack:
                x, y = stack.pop()
                group.append((x, y))
                for nx, ny in ((x + 1, y), (x - 1, y), (x, y + 1), (x, y - 1)):
                    if 0 <= nx < n and 0 <= ny < m and a[nx][ny] == '.':
                        a[nx][ny] = '*'
                        stack.append((nx, ny))

            groups.append(group)

    def func(group):
        rows = [p[0] for p in group]
        cols = [p[1] for p in group]

        min_r, max_r = min(rows), max(rows)
        min_c, max_c = min(cols), max(cols)

        height = max_r - min_r + 1
        width = max_c - min_c + 1

        return len(group) == height * width

    ans = 0
    for group in groups:
        ans += func(group)

    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
