**思路：随机化。硬找精确点很困难，并且题目给出的n比较小，还一定存在答案。因此考虑随机化。直接随机出一个二维点，然后判断是否在这n个探测器之外即可**

**代码：**
```python
import random
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''


def solve():
    n, l = MII()
    circles = [tuple(MII()) for _ in range(n)]

    def safe(px, py):
        for x, y, r in circles:
            dx, dy = px - x, py - y
            if dx * dx + dy * dy <= r * r + 1e-9:
                return False
        return True

    for p in [(0, 0), (0, l), (l, 0), (l, l)]:
        if safe(*p):
            print(f"{p[0]:.10f} {p[1]:.10f}")
            return

    random.seed(0)
    for _ in range(2 * 10 ** 6):
        x = random.random() * l
        y = random.random() * l
        if safe(x, y):
            print(f"{x:.10f} {y:.10f}")
            return

t = 1
# t = II()
for _ in range(t):
    solve()
```
