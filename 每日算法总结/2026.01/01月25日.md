## 第二题：Permutation² Counting https://ac.nowcoder.com/acm/contest/126798/D

**思路：思路不难想，但是我代码里面的for判断写得有问题，不能直接print + return，如果遇到1 1 2 2这种数据，就不会有结果输出，导致出错！所以今后不要再出现这种写法，调逻辑调爆了才发现是这个返回值问题**

**代码：**
```python
import sys
from collections import Counter

input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

mod = 998_244_353

def solve():
    n = II()
    a = LII()
    cnt = Counter(a)
    ans = 0
    pre_s = 1
    pre_k = 0
    for k, v in sorted(cnt.items()):
        if v < 2 or pre_k + 1 != k:
            break
        s = (pre_s * v * (v - 1) % mod) * pow(2, mod - 2, mod) % mod
        ans = (ans + s) % mod
        pre_s = s
        pre_k = k
    print(ans)

# t = 1
t = II()
for _ in range(t):
    solve()
```
