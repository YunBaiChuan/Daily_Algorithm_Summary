## 第一题：dd爱冒险 https://www.nowcoder.com/practice/e97ab78badf64bbf947fe2097d5de355?channelPut=tracker2

**思路：要恰好走n - 1条路，把所有的点都给经过，这其实是一个旅行商问题，或者说是一个哈密顿路径问题。这种问题的n一般都很小，因为他的时间复杂度起手就是$2^n$级别，我们要用状压dp来做，比较难，先放着**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n = II()
    p = [LII() for _ in range(n)]

    # dp[mask][i][g][a]  最小代价
    SIZE = 1 << n
    dp = [[[[inf]*2 for _ in range(2)] for __ in range(n)] for ___ in range(SIZE)]
    for i in range(n):
        dp[1 << i][i][0][0] = 0

    for mask in range(SIZE):
        for i in range(n):
            if not (mask >> i & 1):
                continue
            for g in range(2):
                for a in range(2):
                    cur = dp[mask][i][g][a]
                    if cur == inf:
                        continue
                    # 走到 j
                    for j in range(n):
                        if mask >> j & 1:
                            continue
                        w = p[i][j]
                        nmask = mask | (1 << j)
                        # 1. 正常走
                        if cur + w < dp[nmask][j][g][a]:
                            dp[nmask][j][g][a] = cur + w
                        # 2. 重力加速
                        if g == 0 and cur < dp[nmask][j][1][a]:
                            dp[nmask][j][1][a] = cur
                        # 3. 反重力加速
                        if a == 0 and cur + 2 * w < dp[nmask][j][g][1]:
                            dp[nmask][j][g][1] = cur + 2 * w

    full = (1 << n) - 1
    final_ans = inf
    for i in range(n):
        final_ans = fmin(final_ans, dp[full][i][1][1])
    print(final_ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```
