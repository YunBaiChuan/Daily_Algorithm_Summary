import math

def I():

def LII():

def solve():
    a = list(map(int, I()))

    l = len(a)
    pre = 0
    for i in range(2, -1, -1):
        a[i] += pre
        if a[i] >= 5 and i == 2 or a[i] == 10 and i != 2:
            pre = 1
            if a[i] == 10:
                a[i] = 0
        else:
            pre = 0
    if pre == 1 and a[0] == 0:
        a = [1] + a
        l += 1

    print(f"{a[0]}.{a[1]}*10^{l - 1}")

t = 1
