**思路：预处理 + 分组循环。对于交替这个要求，从字符的角度来说不好处理，因此我们可以把目标字符转化为数字0和1，然后再用分组循环的异或，因为异或的^1性质非常适合两个字符的交替，最终不断取长度的max就是答案**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

def solve():
    n = II()
    s = I()

    a = [-1] * n
    for i, c in enumerate(s):
        if c == 'a':
            a[i] = 0
        elif c == 'h':
            a[i] = 1

    ans = 0
    i = 0
    while i < n:
        if a[i] == -1:
            i += 1
            continue
        st = i
        i += 1
        while i < n and a[i] == a[i - 1] ^ 1:
            i += 1
        ans = max(ans, i - st)
    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```
