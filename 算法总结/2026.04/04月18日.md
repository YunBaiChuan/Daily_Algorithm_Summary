import bisect
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

def solve():
    n = II()
    a = [0] + LII() 

    p_input = LII()
    p = [0] * (n + 1)

    if len(p_input) == n:
        for i in range(1, n + 1):
            p[i] = p_input[i - 1]
    else:
        p[1] = 0
        for i in range(2, n + 1):
            p[i] = p_input[i - 2]

    g = [[] for _ in range(n + 1)]
    for i in range(2, n + 1):
        g[p[i]].append(i)

    pre = [0] * (n + 1)
    sub = [0] * (n + 1)

    for i in range(1, n + 1):
        pre[i] = pre[p[i]] + a[i]
        sub[i] = a[i]


    for i in range(n, 1, -1):
        sub[p[i]] += sub[i]

    c1 = [False] * (n + 1)
    c2 = [False] * (n + 1)
    sup = [False] * (n + 1)
    sp = [False] * (n + 1)
    nd = [0] * (n + 1)
    vals = []
    base = 0

    for i in range(1, n + 1):
        c1[i] = (pre[i] >= 2 * a[i])
        c2[i] = (sub[i] <= 2 * a[i])
        sup[i] = (c1[i] and c2[i])
        base += 1 if sup[i] else 0
        if c1[i] and not c2[i]:
            sp[i] = True
            nd[i] = sub[i] - 2 * a[i]
            vals.append(nd[i])

    cnt = [0] * (n + 1)
    for i in range(1, n + 1):
        cnt[i] = 1 if sup[i] else 0
    for i in range(n, 1, -1):
        cnt[p[i]] += cnt[i]

    vals = sorted(set(vals))
    m = len(vals)

    bit = [0] * (m + 2)

    def add(x, v):
        while x <= m:
            bit[x] += v
            x += x & -x

    def ask(x):
        s = 0
        while x > 0:
            s += bit[x]
            x -= x & -x
        return s

    def get_id(x):
        pos = bisect.bisect_left(vals, x)
        return pos + 1

    def get_le(x):
        pos = bisect.bisect_right(vals, x)
        return pos

    gain = [0] * (n + 1)
    st = []
    op = []

    st.append(1)
    op.append(0)

    while st:
        u = st.pop()
        t = op.pop()
        if t == 0:
            gain[u] = ask(get_le(sub[u]))
            if sp[u]:
                add(get_id(nd[u]), 1)
            st.append(u)
            op.append(1)
            for v in reversed(g[u]):
                st.append(v)
                op.append(0)
        else:
            if sp[u]:
                add(get_id(nd[u]), -1)

    ans = base
    for v in range(2, n + 1):
        cur = base - cnt[v] + gain[v]
        ans = fmax(ans, cur)
    print(ans)


t = 1
# t = II()
for _ in range(t):
    solve()
