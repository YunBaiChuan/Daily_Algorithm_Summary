**思路：这题不需要开第三个维度，我们只需要提前标记无法访问的格子为-inf，这样即使后续访问到了标记为-inf格子，他也不可能成为最终的答案，这个做法很巧妙。标记的话，其实就是一个曼哈顿距离，即横坐标与纵坐标之和，这点我也没想到**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, m = MII()
    grid = []
    for _ in range(n):
        grid.append(LII())

    t = II()
    for _ in range(t):
        x, y, v = MII()
        if (x - 1 + y - 1 >= v):
            grid[x - 1][y - 1] = -inf

    ans = grid[0][0]

    for j in range(1, m):
        grid[0][j] += grid[0][j - 1]
    for i in range(1, n):
        grid[i][0] += grid[i - 1][0]

    for i in range(1, n):
        for j in range(1, m):
            grid[i][j] += fmax(grid[i - 1][j], grid[i][j - 1])
            ans = fmax(ans, grid[i][j])

    print(ans)

tt = 1
# t = II()
for _ in range(tt):
    solve()
```
