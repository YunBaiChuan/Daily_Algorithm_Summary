class UnionFind:
    def __init__(self, n):
        self.fa = list(range(n))
        self.size = [1] * n
        self.sets = n

    def find(self, x):
        if x != self.fa[x]:
            self.fa[x] = self.find(self.fa[x])
        return self.fa[x]

    def is_same(self, x, y):
        return self.find(x) == self.find(y)

    def union(self, x, y):
        fx, fy = self.find(x), self.find(y)
        if fx == fy:
            return False
        self.fa[fx] = fy
        self.size[fy] += self.size[fx]
        self.sets -= 1
        return True

def solve():
    n = II()
    a = [0] + LII()
    uf = UnionFind(n + 1)
    for _ in range(n - 1):
        u, v = MII()
        if not uf.is_same(u, v) and a[u] == a[v]:
            uf.union(u, v)

    ans = []
    mx = max(uf.size)
    for i in range(1, n + 1):
        fi = uf.find(i)
        if uf.size[fi] == mx:
            ans.append(i)

    print(len(ans))
    print(*ans)

    solve()
