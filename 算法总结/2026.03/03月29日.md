import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, a, b = LII()

    def cal(x, y):
        res = n
        while x >= a or y >= b:
            res += x // a + y // b
            x, y = (x % a) + y // b, (y % b) + x // a
        return res

    ans = 0
    for x in range(n + 1):
        ans = fmax(ans, cal(x, n - x))

    print(ans)

# t = 1
t = II()
for _ in range(t):
    solve()
