out = []

def solve():
    p, q = MII()

    def func(p, q):
        a = []
        while q:
            div = p // q
            rem = p % q
            a.append(div)
            p, q = q, rem
        return a

    def output(p, q, a):
        n = len(a) - 1
        if n == 0:
            return f"{p}/{q} = {a[0]}"
        res = f"{p}/{q} = {a[0]}+1/"
        for i in range(1, n):
            res += f"{{{a[i]}+1/"
        res += f"{a[n]}"
        res += "}" * (n - 1)
        return res

    a = func(p, q)
    out.append(output(p, q, a))

# t = 1
t = II()
for _ in range(t):
    solve()

print('\n'.join(out))
