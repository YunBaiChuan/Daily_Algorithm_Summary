**思路：暴力枚举列m，从增加列到减少列，枚举完之后就按列进行遍历，判断当前列中最小的，全部变为k的操作次数，然后再累加列的变换次数，就得到当前答案。最终，不断取min就得到最后的答案了**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def num_cost(new_m, n, k, nums):
    now_ans = inf
    for i in range(new_m):
        one = 0
        j = i
        while j < n:
            if nums[j] != k:
                one += 1
            j += new_m
        now_ans = fmin(now_ans, one)
    return now_ans


def solve():
    n, m, k = MII()
    nums = LII()

    ans = inf

    i = m
    while i <= n and i - m < ans:
        ans = fmin(ans, num_cost(i, n, k, nums) + i - m)
        i += 1

    i = m - 1
    while i > 0 and m - i < ans:
        ans = fmin(ans, num_cost(i, n, k, nums) + m - i)
        i -= 1

    print(ans)


# t = 1
t = II()
for _ in range(t):
    solve()
```
