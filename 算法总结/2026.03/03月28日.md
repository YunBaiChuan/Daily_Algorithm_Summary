**思路：欧式筛/线性筛，其时间复杂度为$O(n)$，不过常数时间比较大一些。其具体原理如下：对于每个数来说，每次乘一个小于等于他最小质因子的质数，就可以划掉一个相应的合数，并且每个合数只会被划掉一次，因此其时间复杂度就是线性的$O(n)$**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
欧式筛
'''

def solve():
    n = II()
    ans = 0
    is_prime = [True] * (n + 1)
    primes = []
    for i in range(2, n + 1):
        if is_prime[i]:
            ans += i
            primes.append(i)
        for p in primes:
            if p * i > n:
                break
            is_prime[p * i] = False
            ans += p
            if i % p == 0:
                break
    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```
