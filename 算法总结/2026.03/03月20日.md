**思路：从高位向低位进行贪心，然后需要额外判断一下当前位置无法分配，但是有多余的情况，我们可以把多余的就放在mask这一位，然后在n中减去他，这也算是一种贪心策略**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y

isqrt = lambda x: int(math.sqrt(x))

def solve():
    n, m = MII()
    ans = 0
    for i in range(31, -1, -1):
        mask = 1 << i
        if n >= mask * m:
            n -= mask * m
            ans |= mask
        elif n > mask * m - m:
            n -= (n - mask * m + m + mask - 1) // mask * mask
    print(ans)

# t = 1
t = II()
for _ in range(t):
    solve()
```
