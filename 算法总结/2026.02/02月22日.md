**思路：状态机dp，先放着，目前不是很懂他的逻辑。其实这个数据也可以用枚举，枚举三个边界部分，然后处理出111000111000和000111000111这两种类型需要的操作次数，最终取fmin即可**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n = II()
    s = list(map(int, I()))

    dp = [[inf] * 2 for _ in range(n + 1)]
    dp[0][s[0]] = 0
    dp[0][s[0] ^ 1] = 1

    for i in range(1, n):
        new_dp = [[inf] * 2 for _ in range(n + 1)]
        for c in range(i + 1):
            for v in [0, 1]:
                if dp[c][v] == inf:
                    continue
                for nv in [0, 1]:
                    nc = c + (v ^ nv)
                    if nc > n:
                        continue
                    cost = dp[c][v] + (nv ^ s[i])
                    new_dp[nc][nv] = fmin(new_dp[nc][nv], cost)
        dp = new_dp

    ans = fmin(dp[3][0], dp[3][1])
    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```
