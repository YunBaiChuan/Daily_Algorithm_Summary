import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y

def solve():
    MOD = 10 ** 9 + 7
    n = II()
    a = LII()

    if n == 1:
        res = [0] * 10
        if 0 <= a[0] <= 9:
            res[a[0]] = 1
        print(*res)
        return

    dp = [0] * 10
    dp[a[n - 1] % 10] = 1

    for i in range(n - 2, -1, -1):
        nxt = [0] * 10
        cur = a[i] % 10
        for j in range(10):
            if dp[j]:
                nxt[(cur + j) % 10] = (nxt[(cur + j) % 10] + dp[j]) % MOD
                nxt[(cur * j) % 10] = (nxt[(cur * j) % 10] + dp[j]) % MOD
        dp = nxt

    print(*dp)

t = 1
# t = II()
for _ in range(t):
    solve()
