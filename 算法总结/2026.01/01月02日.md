def solve():
    n, k = MII()
    a = LII()
    
    def check(mid):
        cnt = 0
        for i in range(n):
            cnt += a[i] // mid
        return cnt >= k
    
    l, r = 1, 10 ** 9
    while l <= r:
        mid = (l + r) >> 1
        if check(mid):
            l = mid + 1
