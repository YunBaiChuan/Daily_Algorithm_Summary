**思路：用dp打表找规律，可以发现在x > 14时，结果都为True；在x <= 14时，结果不都为True。那么我们就在x <= 14的时候用dp表判断一下，否则直接输出Yes即可。这个dp得研究下，看是什么类型**

def dp_table(max_n):
    dp = [False] * (max_n + 1)
    dp[1] = True
    for i in range(2, max_n + 1):
        k = 1
        while k * k <= i:
            j = i - k * k + 1
            if 1 <= j <= i and dp[j]:
                dp[i] = True
                break
            k += 1
    return dp

dp = dp_table(100)

# for i in range(1, 101):

def solve():
    x = II()
    if x <= 14:
        print("Yes" if dp[x] else "No")
    else:
        print("Yes")

# t = 1
t = II()
for _ in range(t):
    solve()
