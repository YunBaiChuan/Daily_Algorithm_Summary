**思路：由于数据量很小，可以考虑$O(n!)$复杂度的算法，这里应该用回溯，不过我没想到。定义$dfs(i, j)$表示上一个字符为i，剩余要填的数量为j，当我们填的字符与i不同时，并且对应的字符数量不为0，才可以填，之后dfs完了要进行现场恢复**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
回溯
'''

def solve():
    s = I()
    cnt = [0] * 26
    for c in s:
        cnt[ord(c) - ord('a')] += 1
    n = len(s)

    def dfs(prev, left): # prev表示上一个填的编号，left表示剩下几个字符要填
        if left == 0:
            return 1
        res = 0
        for i in range(26):
            if i != prev and cnt[i]:
                cnt[i] -= 1
                res += dfs(i, left - 1)
                cnt[i] += 1 # 恢复现场
        return res

    print(dfs(-1, n))

t = 1
for _ in range(t):
    solve()
```
