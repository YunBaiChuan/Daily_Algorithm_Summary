import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n = II()
    a = [None] + [LII() for _ in range(n)]

    # dp[i] 以 i 结尾的最长交替子序列长度
    dp = [1] * (n + 1)

    # t[skill][hand] 记录 skill 固定、hand 状态时的最大 dp 值
    t = [[0, 0] for _ in range(11)]   # skill 1~10

    # 初始化第一个元素
    dp[1] = 1
    t[a[1][1]][a[1][0]] = 1

    for i in range(2, n + 1):
        hand_i, skill_i = a[i]
        # 禁止同一 skill，左右手必须交替
        for skill_j in range(1, 11):
            if skill_j == skill_i:
                continue
            dp[i] = fmax(dp[i], t[skill_j][hand_i ^ 1] + 1)
        # 更新表
        t[skill_i][hand_i] = fmax(t[skill_i][hand_i], dp[i])

    print(max(dp))   # 全局最大值

tt = II()
for _ in range(tt):
    solve()
