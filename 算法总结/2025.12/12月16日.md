**思路：模拟题，算一下时间复杂度可以得到是$O(2^q * n)$。由于q只有10，那么量级就是$2^{10} * 1000$，为$10^6$，完全可以通过**

**代码：**

import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y

def solve():
    n, q = MII()
    s = I()
    for _ in range(q):
        res = ''
        l, r = MII()
        l -= 1
        for i in range(l, r):
            c = s[i]
            res += c * 2
        s = s[:l] + res + s[r:]
