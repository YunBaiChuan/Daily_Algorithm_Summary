**思路：很神秘的一题，数据比较水。所以用经典的子序列匹配算法，也就是双指针，他的时间复杂度是$O(|s|+|t|)$，有n组数据，那么整体时间复杂度就是$O(n * (|s|+|t|))$，他居然可以通过**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def GMI():
    return map(lambda x: int(x) - 1, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
很神秘的题目，按道理来说这个复杂度早超时了，但是能过
'''

def solve():
    a = I()
    n = II()

    def func(s, t):
        p, q = len(s), len(t)
        i = j = 0
        while i < p and j < q:
            if s[i] == t[j]:
                i += 1
                j += 1
            else:
                i += 1
        return j == q

    for _ in range(n):
        b = I()
        if func(a, b):
            print("Yes")
        else:
            print("No")


tt = 1
# t = II()
for _ in range(tt):
    solve()
```
