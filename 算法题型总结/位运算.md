### 1.小苯的01背包 https://www.nowcoder.com/practice/32407155397c443a95f895f11942f02f?channelPut=tracker2

**思路：这题不能用01背包板子，因为&没有+那样的单调性，状态转移写不出来。因此，本题需要用位枚举，从高位到低位进行枚举，然后判断能够达到答案**

**代码：**
```python
import sys
input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

def solve():
    n, k = MII()
    a = [LII() for _ in range(n)]

    ans = 0
    for mask in range(10, -1, -1):
        target = ans | (1 << mask)
        v_and = (1 << 11) - 1
        ok = False

        for i in range(n):
            v, w = a[i]
            if w & target == target: # 能选的都选，可以把体积变小，并且当前位的结果依然最优
                v_and &= v
                ok = True

        if ok and v_and <= k:
            ans = target

    print(ans)

t = 1
# t = II()
for _ in range(t):
    solve()
```

### 2.统计节点和为偶数的连通子图 https://leetcode.cn/problems/count-connected-subgraphs-with-even-node-sum/description/

**思路：位运算枚举子集 + dfs判断连通性。这个要进行局部的枚举子集的连通性判断，不能用并查集全局连通性判断，对于本题来说，用dfs判断连通性写起来比较简单**

**代码：**
```python
class Solution:
    def evenSumSubgraphs(self, nums: list[int], edges: list[list[int]]) -> int:
        n = len(nums)
        g = [[] for _ in range(n)]
        for x, y in edges:
            g[x].append(y)
            g[y].append(x)

        u = (1 << n) - 1
        ans = 0

        for sub in range(1, u + 1):
            xor_sum = 0
            for i, x in enumerate(nums):
                if sub >> i & 1:
                    xor_sum ^= x
            if xor_sum:
                continue

            def dfs(x: int) -> None:
                nonlocal vis
                vis |= 1 << x
                for y in g[x]:
                    if (vis >> y & 1) == 0:
                        dfs(y)

            vis = u ^ sub
            dfs(sub.bit_length() - 1)
            if vis == u:
                ans += 1

        return ans
```

### 3.出题需要加法 https://ac.nowcoder.com/acm/contest/134527/C

**思路：位运算。首先由题意可知，满足条件的数要么是二进制中有两个1，要么是二进制中有一个1，并且值不为1；所以说满足条件的数是很少的，我们就可以提前预处理出所有满足条件的数，并且将这些数进行排序；最后，利用二分查找优化，算出$[l, r]$区间中，有多少个满足条件的数**

**代码：**
```python
import sys
from bisect import bisect_right, bisect_left

input = lambda: sys.stdin.readline().strip()

import math
inf = 10 ** 18

def I():
    return input()

def II():
    return int(input())

def MII():
    return map(int, input().split())

def LI():
    return input().split()

def LII():
    return list(map(int, input().split()))

def LFI():
    return list(map(float, input().split()))

fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''

'''

ans = set()
cur = 0
for i in range(64):
    cur |= 1 << i
    pre = cur
    for j in range(64):
        cur |= 1 << j
        if 0 <= cur < pow(2, 63):
            ans.add(cur)
        cur = pre
    cur = 0

if 1 in ans:
    ans.remove(1)
ans = sorted(ans)

def solve():
    l, r = MII()
    print(bisect_right(ans, r) - bisect_left(ans, l))

# t = 1
t = II()
for _ in range(t):
    solve()
```
